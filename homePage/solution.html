function solution(arr){			
	//define two varibles to record the maximum index and secondeary index,
	//and then loop through the array.
	var first=arr[0]>=arr[1]?0:1,
		second=arr[0]>=arr[1]?1:0,
		j=2;				
	while(j<arr.length){
		if(arr[j]>arr[first]){
			second=first;
			first=j;
		}else if(arr[j]>arr[second]){
			second=j;
		}
		j++;
	}
	return arr[first]*arr[second];		
}
